func setupUI() {
    self.remoteUserView = UIView(frame: CGRect(x: 100, y: 100, width: 180, height: 250))
    self.remoteUserView.backgroundColor = UIColor.lightGray
    self.view.addSubview(self.remoteUserView)
    
    self.startVideoTalkButton = UIButton(type: .system)
    self.view.addSubview(self.startVideoTalkButton)
    self.startVideoTalkButton.frame = CGRect(x: 100, y: self.view.bounds.size.height - 280, width: 150, height: 50)
    self.startVideoTalkButton.titleLabel?.font = UIFont.systemFont(ofSize: 32)
    self.startVideoTalkButton.setTitle("Start", for: .normal)
    self.startVideoTalkButton.addTarget(self, action: #selector(startVideoTalk(_:)), for: .touchUpInside)

    self.view.addSubview(self.stopVideoTalkButton)

    self.stopVideoTalkButton.setTitle("Stop", for: .normal)
    self.stopVideoTalkButton.setTitleColor(UIColor.red, for: .normal)
